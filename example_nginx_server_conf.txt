# Example NGINX reverse proxy configuration for Abyss server operating on default port 1566
server {
    server_name discord.mydomain.com;
    listen 80 http2;
    listen [::]:80 http2;

    listen 443 ssl http2;
    listen [::]:443 ssl http2;

    # Point these to your SSL certificates (for example, from Cloudflare if you're using the SSL Strict option)
    ssl_certificate /etc/nginx/ssl/mydomain-cert.pem;
    ssl_certificate_key /etc/nginx/ssl/mydomain-key.pem;

    location / {
        proxy_pass          http://localhost:1566;
        proxy_http_version  1.1;
        proxy_set_header    Upgrade $http_upgrade;
        proxy_set_header    Connection keep-alive;
        proxy_set_header    Host $host;
        proxy_cache_bypass  $http_upgrade;
        proxy_set_header    X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header    X-Forwarded-Proto $scheme;
    }
}
